### 5.2 自动化测试脚本开发

#### 5.2.1 导入路径验证脚本

```python
# scripts/tools/verify_imports.py
import sys
import importlib
from typing import List, Tuple

def verify_imports() -> List[Tuple[str, bool, str]]:
    """验证所有模块导入"""
    
    imports_to_test = [
        ("src.core.code_generator", "generate_code"),
        ("src.core.input_parser", "parse_user_input"),
        ("src.core.text_chunk", "chunk_text"),
        ("src.llm.local_llm_config", "LocalLLMConfig"),
        ("src.llm.ollama_api", "call_ollama_api"),
        ("src.evaluation.ragas_with_langsmith", None),
        ("src.evaluation.environment_validator", "validate_environment"),
        ("src.utils.retry_decider", "should_retry"),
    ]
    
    results = []
    
    for module_name, function_name in imports_to_test:
        try:
            module = importlib.import_module(module_name)
            if function_name:
                getattr(module, function_name)
            results.append((module_name, True, "导入成功"))
        except Exception as e:
            results.append((module_name, False, str(e)))
```

```python
import datetime
from pathlib import Path

def generate_test_report():
    """生成测试报告"""
    
    report_date = datetime.datetime.now().strftime("%Y-%m-%d")
    report_file = f"docs/reports/test_report_{report_date}.md"
    
    # 收集测试数据
    test_results = collect_test_results()
    coverage_data = collect_coverage_data()
    performance_data = collect_performance_data()
    issues = collect_issues()
    
    # 生成报告
    report_content = f"""
# 项目文件整理后功能验证测试报告

## 1. 测试概述

验证项目文件整理后系统功能的完整性和正确性

- 测试日期: {report_date}

## 2. 测试结果

### 2.1 测试用例执行情况

{generate_test_results_table(test_results)}

### 2.2 测试覆盖率

{generate_coverage_table(coverage_data)}

### 2.3 性能测试结果

{generate_performance_table(performance_data)}

## 3. 问题统计

{generate_issues_table(issues)}

## 4. 测试结论

{generate_conclusion(test_results, coverage_data, performance_data)}
"""
    
    # 写入报告文件
```
